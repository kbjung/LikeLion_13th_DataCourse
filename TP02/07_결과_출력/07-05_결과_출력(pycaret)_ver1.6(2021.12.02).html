    temp_train = train[train['구분'] == division].reset_index(drop=True)
    temp_test = test2019[test2019['구분'] == division].reset_index(drop=True)

    # 해당일 공급량 = 전날 공급량 * ( 1 + 해당일 공급량 비율 )
